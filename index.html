Algorithme Suite Arithmétique

tdo:
    n: entier          
    premierTerme: réel  
    raison: réel   
    i: entier           
    terme: réel         

Début

    ecrire("Entrez le nombre de termes à affchier (n): ") lire(n)
    ecrire("Entrez le premier terme de la suite: ") lire(premierTerme)
    ecrire("Entrez la raison de la suite: ") lire(raison)

    ecrire("Les ", n, " premiers termes de la suite sont:")

    pour i de 0 à n-1 faire
        terme ← premierTerme + i * raison
        ecrie(terme)
    fin_pour 

Fin

Algorithme Vérifier Code

tdo:
    code: chaîne
    erreur: entier
    i: entier

début
    ecrie("entree le code") lire(code)

    si long(code-1) ≠ 9 alors
        ecrie("Code invalide: doit avoir 9 caractères")
    fin_si

    sinon si code[0] < '0' or code[0] > '9' alors
        ecrie("Code invalide: le premier caractère doit être un chiffre")
    fin_si

    sinon si code[1] != '-' alors
        ecrie("Code invalide: le deuxième caractère doit être '-'")
    fin_si

    sinon
        erreur = 0
        pour i de 2 à 8 faire
            si code[i] < 'A' or code[i] > 'Z' alors
                erreur = 1
            fin_si
        fin_pour

        si erreur = 1 alors
            ecrie("Code invalide: les caractères doivent être des lettres majuscules")
        sinon
            ecrire("Code valide")
        fin_si
    fin
